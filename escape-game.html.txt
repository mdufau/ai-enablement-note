        const all = document.querySelectorAll('#choices-' + room + ' .choice-btn');
        all.forEach(b => b.classList.add('disabled'));
        if (isCorrect) {
            btn.classList.remove('disabled');
            btn.classList.add('correct');
            score++;
        }
        else {
            btn.classList.remove('disabled');
            btn.classList.add('wrong');
            all.forEach(b => {
                if (b.getAttribute('onclick').includes(',true')) {
                    {
                        b.classList.remove('disabled');
                        b.classList.add('reveal-correct');
                    }
                }
            });
        }
        const fb = document.getElementById('feedback-' + room);
        fb.className = 'feedback ' + (isCorrect ? 'show-ok' : 'show-ko');
        fb.innerHTML = '<div class="fb-label">' + (isCorrect ? '‚úì La porte s\'ouvre' : '‚úó La porte r√©siste') + '</div>' + FB[room][isCorrect ? 'ok' : 'ko'];
        if (isCorrect)
            setTimeout(() => document.getElementById('unlock-' + room).classList.add('show'), 420);
        setTimeout(() => document.getElementById('next-' + room).classList.add('show'), isCorrect ? 1300 : 650);
    }

    function getElapsed() {
        const s = Math.round((Date.now() - startTime) / 1000);
        return s < 60 ? s + 's' : Math.floor(s / 60) + 'min ' + (s % 60) + 's';
    }

    function showResult() {
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
        document.getElementById('screen-result').classList.add('active');
        document.getElementById('progressFill').style.width = '100%';
        document.getElementById('progressText').textContent = 'Termin√©';
        document.getElementById('finalScore').innerHTML = score + '<span>/4</span>';
        document.getElementById('rc-correct').textContent = score + '/4';
        document.getElementById('rc-time').textContent = getElapsed();
        const v = VERDICTS.find(x => score >= x.min && score <= x.max);
        document.getElementById('resultInsight').innerHTML = '<strong>Analyse :</strong> ' + v.txt;
        window.scrollTo({
            top: 0,
            behavior: 'smooth'
        });
    }

    }

    function restartGame() {
        score = 0;
        answered = {};
        for (let r = 1; r <= 4; r++) {
            document.querySelectorAll('#choices-' + r + ' .choice-btn').forEach(b => b.classList.remove('correct', 'wrong', 'reveal-correct', 'disabled'));
            const fb = document.getElementById('feedback-' + r);
            if (fb) {
                fb.className = 'feedback';
                fb.innerHTML = '';
            }
            const nb = document.getElementById('next-' + r);
            if (nb)
                nb.classList.remove('show');
            const ul = document.getElementById('unlock-' + r);
            if (ul)
                ul.classList.remove('show');
        }
        document.getElementById('progressArea').classList.remove('show');
        document.getElementById('progressFill').style.width = '0%';
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
        document.getElementById('screen-intro').classList.add('active');
        window.scrollTo({
            top: 0,
            behavior: 'smooth'
        });
    }
